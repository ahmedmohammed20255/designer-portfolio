const dialog = document.getElementById('product-dialog');
const priceElement = document.getElementById('price-1');

priceElement.addEventListener('click', () => {
dialog.showModal();
});

window.closeDialog = function() {
dialog.close();
};
// Handle category clicks
const categories = document.querySelectorAll('.grid-cols-3 > div');
categories.forEach(category => {
category.addEventListener('click', function() {
// Navigate to category page (placeholder)
console.log('Category clicked:', this.querySelector('span').textContent);
});
});
// Handle product clicks
const products = document.querySelectorAll('.rounded-lg.shadow-sm.overflow-hidden.cursor-pointer');
products.forEach(product => {
product.addEventListener('click', function() {
// Navigate to product details page (placeholder)
console.log('Product clicked:', this.querySelector('h3').textContent);
});
});
// Handle add to cart buttons
const addButtons = document.querySelectorAll('.ri-add-line');
addButtons.forEach(button => {
button.addEventListener('click', function(e) {
e.stopPropagation();
// Add to cart functionality (placeholder)
const productName = this.closest('.p-3').querySelector('h3').textContent;
console.log('Added to cart:', productName);
// Show mini notification
const notification = document.createElement('div');
notification.className = 'fixed top-16 left-1/2 transform -translate-x-1/2 bg-green-500 text-white px-4 py-2 rounded-lg shadow-lg z-50';
notification.textContent = 'تمت الإضافة إلى السلة';
document.body.appendChild(notification);
setTimeout(() => {
notification.remove();
}, 2000);
});
});
// Handle custom design button
const customDesignBtn = document.querySelector('button.w-full.py-3.bg-primary');
customDesignBtn.addEventListener('click', function() {
// Navigate to custom design page (placeholder)
console.log('Custom design requested');
});
// Handle chat button
const chatButton = document.querySelector('.fixed.right-4.bottom-24 button');
chatButton.addEventListener('click', function() {
// Open chat interface (placeholder)
console.log('Chat button clicked');
});
// Handle tab bar navigation
const tabItems = document.querySelectorAll('.grid-cols-5 > div');
tabItems.forEach((item, index) => {
item.addEventListener('click', function() {
// Reset all tabs
tabItems.forEach(tab => {
const icon = tab.querySelector('i');
const text = tab.querySelector('span');
icon.className = icon.className.replace('-fill', '-line').replace('text-primary', 'text-gray-500');
text.className = text.className.replace('text-primary', 'text-gray-500');
});
// Set active tab
const icon = this.querySelector('i');
const text = this.querySelector('span');
icon.className = icon.className.replace('-line', '-fill').replace('text-gray-500', 'text-primary');
text.className = text.className.replace('text-gray-500', 'text-primary');
// Navigate to tab (placeholder)
const tabNames = ['الرئيسية', 'الخدمات', 'السلة', 'الطلبات', 'حسابي'];
console.log('Tab clicked:', tabNames[index]);
});
});
});
</script>
</body>
</html>
